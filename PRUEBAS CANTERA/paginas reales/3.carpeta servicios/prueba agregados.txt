        document.addEventListener('DOMContentLoaded', function() {
            const background1 = document.getElementById('background1');
            const background2 = document.getElementById('background2');
            const images = [
                'url("fotico.jpg")',
                'url("fondo dos.jpg")',
                'url("fondo 3.jpg")',
            ];
            let currentIndex = 0;
            let showingBackground1 = true;

            function changeBackground() {
                currentIndex = (currentIndex + 1) % images.length;
                if (showingBackground1) {
                    background2.style.backgroundImage = images[currentIndex];
                    background2.classList.add('show');
                    background1.classList.remove('show');
                } else {
                    background1.style.backgroundImage = images[currentIndex];
                    background1.classList.add('show');
                    background2.classList.remove('show');
                }
                showingBackground1 = !showingBackground1;
            }

            setInterval(changeBackground, 5000); // Cambia la imagen cada 5 segundos

            // Inicializa la primera imagen
            background1.style.backgroundImage = images[0];
            background1.classList.add('show');
        });
    </script>

<script>
    const carouselContainer = document.getElementById('carousel-container');
    const prevButton = document.getElementById('prev');
    const nextButton = document.getElementById('next');

    let currentIndex = 0;

    nextButton.addEventListener('click', () => {
        if (currentIndex < carouselContainer.children.length - 3) {
            currentIndex++;
        } else {
            currentIndex = 0;
        }
        updateCarousel();
    });

    prevButton.addEventListener('click', () => {
        if (currentIndex > 0) {
            currentIndex--;
        } else {
            currentIndex = carouselContainer.children.length - 3;
        }
        updateCarousel();
    });

    function updateCarousel() {
        const width = carouselContainer.children[0].getBoundingClientRect().width;
        carouselContainer.style.transform = `translateX(-${currentIndex * width}px)`;
    }
</script>
